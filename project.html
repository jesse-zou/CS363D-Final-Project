# Data Visualization Helper Function
def box_plot(datasets: Dict[Data, pd.DataFrame], col_name: str, ax: axes.Axes=None) -> None:
    """
    Given a dictionary of DataFrames and a column to compare, loops through each and displays:
        - a box plot comparing the data sets for that column
        - generates any outliers as red squares
        - titles the box plot the name of the column
        - labels each box plot by the data set name
        
    Args:
        datasets: a dictionary that maps names to DataFrames
        col_name: name of the column in the DataFrame to compare
        axs: The axes the box plot should be on. Generate one if None.
    
    Returns:
        None
    """
    
    # Grab the first data set as a baseline and use it to check if every data set has identical columns.
    baseline = next(iter(datasets.values()))
    assert(all([len(baseline.columns.intersection(dataset.columns)) == 
                len(dataset.columns) for dataset in datasets.values()]))
    
    # Create the red square marker for outliers. 
    red_square = dict(markerfacecolor='r', marker='s')
    
    if ax is None:
        fig, ax = plt.subplots()
    
    ax.set_title(col_name)
    
    # Grab the entire column of each data set and boxplot each.
    data = []
    for dataset in datasets.values():
        data.append(dataset[col_name])
    ax.boxplot(data, vert=False, flierprops=red_square)
    
    # Set the plot's current axes to ax so that we can modify ax through plt.
    plt.sca(ax)
    
    # Update the y tick labels to be the names of the data sets.
    # Need to offset the ticks by 1 since the box plot labels start at 1.
    # Box plots were also added from bottom to top.
    plt.yticks(range(1, len(datasets.values()) + 1), datasets.keys())

def macd(data: pd.DataFrame, period1: int=12, period2: int=26) -> pd.DataFrame:
    """
    Moving average (26 days)
     -  Average price of past N days
        Ex: (MACD = 12 day MA - 26 day MA)
    """
    
    df = data.copy()
    df['p1'] = df['Open'].rolling(period1).mean()
    df['p2'] = df['Open'].rolling(period2).mean()
    data['MACD'] = df['p1'] - df['p2']
    return data

def aroon(data: pd.DataFrame, period: int=25) -> pd.DataFrame:
    """
    Aroon Oscillator (25 days)
    The Aroon Oscillator uses Aroon Up and Aroon Down to create the oscillator.
    Aroon Up and Aroon Down measure the number of periods since the last 25-period high and low.

    """
    
    df = data.copy()
    df['up'] = 100 * df.High.rolling(period + 1).apply(lambda x: x.argmax()) / period
    df['dn'] = 100 * df.Low.rolling(period + 1).apply(lambda x: x.argmin()) / period
    data['Aroon'] = df['up'] - df['dn']
    return data

def ad(data: pd.DataFrame, period: int=1) -> pd.DataFrame:
    """
    Accumulation/Distribution Line (OR Money Flow Volume) (1 Day)
     -  a rising A/D line helps confirm a price uptrend, while a falling A/D line helps confirm a 
        price downtrend.
     -  When the oscillator moves above the zero line it means the Aroon Up is crossing above the Aroon Down. 
        This means the price has made a high more recently than a low. That could be a sign that an uptrend is 
        starting. (Vice versa for moving below zero).
    """
    
    df = data.copy()
    df['MFM'] = df['Volume'] * (((df['Close'] - df['Low']) - (df['High'] - df['Close'])) / (df['High'] - df['Low']))
    data['Accumulation/Distribution'] = df['MFM'].rolling(period).sum()
    return data

def stochastic(data: pd.DataFrame, period: int=14, ma: int=3) -> pd.DataFrame:
    """
    Stochastic Oscillator (14 days)
     -  Ranges from 0 to 100, with <20 being oversold asset and >80 being overbought assets
    """
    
    df = data.copy()
    df['14-high'] = df['High'].rolling(period).max()
    df['14-low'] = df['Low'].rolling(period).min()
    df['%K'] = 100 * (df['Close'] - df['14-low']) / (df['14-high'] - df['14-low'])
    data['Stochastic Oscillator'] = df['%K'].rolling(ma).mean()
    return data

add_engineered_features.append(stochastic)

The SVM model performs 3 things:

Using these steps, we can mitigate the time spent training the SVM model without severely impacting the accuracy.

def SVM_trainer() -> GridSearchCV:
    """
    Creates a 5 fold cross validation support vector machine classifier.
    
    Args:
        None
    
    Returns:
        A support vector machine classifier as a grid search attempting multiple hyperparameters.
    """
    
    # Creates a scaler, PCA, and SVM classifier.
    scaler = StandardScaler()
    pca = PCA()
    svm = SVC()

    # Pipelines the 3 actions so that the data is first scaled, the reduced in dimensionality, then classified.
    pipeline = Pipeline(steps=[('scaler', scaler), ('pca', pca), ('svm', svm)])

    # Parameters grid that dictates how many dimensions PCA should reduce to 
    # and what type of kernel SVM should use.
    param_grid = {
        'pca__n_components': list(range(1, 11)),
        'svm__kernel': ['linear', 'rbf', 'poly']
    }

    # Runs 5 cross fold validation along all hyperparameters in the grid for the pipeline.
    grid_search = GridSearchCV(pipeline, param_grid, cv=5, scoring='accuracy')
    
    return grid_search

def feature_engineer_2(datasets: Dict[str, pd.DataFrame]
                      ) -> Tuple[Dict[str, pd.DataFrame], Dict[str, pd.DataFrame]]:
    """
    Formats the data sets based on the 2nd feature engineering. Drops Date, TEDSpread, and EFFR, then
    separates the features from the labels for each data set.
    
    Args: 
        datasets: a dictionary that maps names to DataFrames
    
    Returns:
        A tuple containing a dictionary of the data set features and a dictionary of the data set labels.
    
    """
    
    fe_features = {}
    fe_labels = {}
    dataset_labels = {}
    
    # Isolates the labels of each data set.
    for name, dataset in datasets.items():
        dataset_labels[name] = dataset.iloc[:, 1]
        
    # Loop through each data set
    for name, dataset in datasets.items():
        
        # Drop specified labels
        processed_dataset = dataset.drop([Features.DATE, Features.TED, Features.EFFR], axis=1)
        
        features = processed_dataset
        
        # Loops through each of the data set labels.
        for other_name, other_dataset_labels in dataset_labels.items():
            
            # Add the data set labels of each of other data sets. Shifts the labels down one
            # so that each data point now also contains the label of the previous data point.
            if name != other_name:
                features[other_name.value] = other_dataset_labels
                features[other_name.value] = features[other_name.value].shift(periods=1, fill_value=-1)
        
        # Drop the first data point since it does not have a previous label
        features.drop(features.index[[0]], inplace=True)
        
        # Separate the features from the labels and then assign them to dictionary based on index name
        labels = features[Features.LABEL]    
        features = features.drop([Features.LABEL], axis=1)
        fe_features[name] = features
        fe_labels[name] = labels
        
    return fe_features, fe_labels

    - The third level dictionary is the mean and standard deviation.
    
    Args:

        None

        fig, axs = plt.subplots(ceil(len(all_formatted_results) / 2), 2,

    for title, formatted_results in all_formatted_results.items():

        if not individual:
            ax = axs[i // 2, i % 2]

        # Set the plot's current axes to ax so that we can modify ax through plt.
        plt.sca(ax)
        
        # Set labels and correct the x axis labels.

        plt.xlabel('Model')
        plt.ylabel('Accuracy (%)')

        plt.legend()
        
        # Increment i so that the correct subplot can be chosen.
        i += 1
    
    # Checks to see if subplots need to be deleted.
    if not individual:
        while i < ceil(len(all_formatted_results) / 2) * 2:
            axs[i // 2, i % 2].set_axis_off()
            i += 1
    
    plt.show()

def format_feature_comparison(fe_results: Dict[str, Dict[str, Dict[str, float]]], 
                              bar_width: int=0.125) -> Dict[str, Dict[str, List[Any]]]:
    """
    Formats the data so that:
    - The first level dictionary are the individual graphs of each index.
    - The second level dictionary are the feature engineerings.
    - The third level list entries is the mean, standard deviation, bar positions, and labels of a model.
    
    Args:
        fe_results: the raw results
        bar_width: width of the bars
    
    Returns:
        The formatted results
    """
    
    all_formatted_results = {}
    
    # Get baselines for the 1st and 2nd level values.
    fe_baseline = next(iter(fe_results.values()))
    index_baseline = next(iter(fe_baseline.values()))
    
    # Because the raw results are ordered as feature engineering -> index -> model results,
    # the formatted results must first create the dictionary levels as:
    # index -> feature engineering -> model results
    # and then iterate through the raw results to append the data.
    for index_name, index in fe_baseline.items():
        # Set initial bar positions.
        bar_positions = np.arange(len(index))
        
        # Create 1st level dictionary entry for each index.
        all_formatted_results[index_name] = {}
        for fe_name, _ in fe_results.items():
            
            # Create 2nd level dictionary entry for each feature engineering.
            all_formatted_results[index_name][fe_name] = {
                "means": [], 
                "stds": [], 
                "positions": bar_positions,
                "models": index.keys()
            }
            
            # Increment bar positions for each subsequent feature engineering.
            bar_positions = [x + bar_width for x in bar_positions]
    
    # Populate list entries by iterating through raw results
    for fe_name, fe in fe_results.items():
        for index_name, index in fe.items():
            for model in index.values():
                all_formatted_results[index_name.value][fe_name]["means"].append(model["mean"])
                all_formatted_results[index_name.value][fe_name]["stds"].append(model["std"])

    return all_formatted_results, bar_positions, bar_width
